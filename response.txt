    <script>
        // Initialize suggestions on page load
        document.addEventListener('DOMContentLoaded', function() {
            loadSuggestions();
        });

        // Tab switching
        function switchTab(tabName) {
            // Hide all tabs
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.classList.remove('active');
            });
            document.querySelectorAll('.tab-button').forEach(btn => {
                btn.classList.remove('active');
            });

            // Show selected tab
            document.getElementById(tabName).classList.add('active');
            event.target.classList.add('active');
        }

        // Load suggestions
        async function loadSuggestions() {
            try {
                const response = await fetch('/api/search');
                const data = await response.json();

                if (data.success) {
                    const suggestionsContainer = document.getElementById('suggestions');
                    suggestionsContainer.innerHTML = '';

                    // Flatten suggestions
                    const allSuggestions = [
                        ...data.suggestions.event_types.slice(0, 3),
                        ...data.suggestions.priorities,
                        ...data.suggestions.components.slice(0, 2)
                    ];

                    allSuggestions.forEach(suggestion => {
                        const item = document.createElement('div');
                        item.className = 'suggestion-item';
                        item.textContent = suggestion;
                        item.onclick = () => insertSuggestion(suggestion);
                        suggestionsContainer.appendChild(item);
                    });
                }
            } catch (error) {
                console.error('Failed to load suggestions:', error);
            }
        }

        // Insert suggestion into query
        function insertSuggestion(suggestion) {
            const textarea = document.getElementById('query-input');
            textarea.value += ` ${suggestion}`;
            textarea.focus();
        }

        // Send single query
        async function sendQuery() {
            const query = document.getElementById('query-input').value.trim();

            if (!query) {
                showError('Please enter a query');
                return;
            }

            if (query.length < 3) {
                showError('Query must be at least 3 characters');
                return;
            }

            showLoading();

            try {
                const response = await fetch('/api/query', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ query })
                });

                const data = await response.json();

                if (data.success) {
                    displayResult(data);
                    hideError();
                } else {
                    showError(data.error || 'Query failed');
                }
            } catch (error) {
                showError(`Error: ${error.message}`);
            } finally {
                hideLoading();
            }
        }

        // Send batch query
        async function sendBatchQuery() {
            const batchText = document.getElementById('batch-queries').value.trim();
            const queries = batchText.split('\n').map(q => q.trim()).filter(q => q.length > 0);

            if (queries.length === 0) {
                showError('Please enter at least one query');
                return;
            }

            if (queries.length > 20) {
                showError('Maximum 20 queries allowed');
                return;
            }

            showLoading();

            try {
                const response = await fetch('/api/batch-query', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ queries })
                });

                const data = await response.json();

                if (data.success) {
                    displayBatchResults(data.results);
                    hideError();
                } else {
                    showError(data.error || 'Batch query failed');
                }
            } catch (error) {
                showError(`Error: ${error.message}`);
            } finally {
                hideLoading();
            }
        }

        // Display single result
        function displayResult(data) {
            const container = document.getElementById('results-container');
            container.innerHTML = '';

            const resultItem = createResultElement(data);
            container.appendChild(resultItem);
            container.style.display = 'flex';

            document.getElementById('empty-state').style.display = 'none';
        }

        // Display batch results
        function displayBatchResults(results) {
            const container = document.getElementById('results-container');
            container.innerHTML = '';

            results.forEach(result => {
                const resultItem = createResultElement(result);
                container.appendChild(resultItem);
            });

            container.style.display = 'flex';
            document.getElementById('empty-state').style.display = 'none';
        }

        // Create result element
        function createResultElement(result) {
            const resultItem = document.createElement('div');
            resultItem.className = 'result-item';

            const qualityClass = `quality-${result.quality || 'unknown'}`;
            const confidencePercent = result.confidence || 0;

            let metadataHTML = '<div class="result-metadata">';

            if (result.components_found && result.components_found.length > 0) {
                metadataHTML += `
                    <div class="metadata-item">
                        <div class="metadata-label">Components</div>
                        <div class="tags">${
                            result.components_found.map(c => `<span class="tag">${c}</span>`).join('')
                        }</div>
                    </div>
                `;
            }

            if (result.priority_levels && result.priority_levels.length > 0) {
                metadataHTML += `
                    <div class="metadata-item">
                        <div class="metadata-label">Priorities</div>
                        <div class="tags">${
                            result.priority_levels.map(p => `<span class="tag">${p}</span>`).join('')
                        }</div>
                    </div>
                `;
            }

            if (result.time_periods && result.time_periods.length > 0) {
                metadataHTML += `
                    <div class="metadata-item">
                        <div class="metadata-label">Time Periods</div>
                        <div class="tags">${
                            result.time_periods.map(t => `<span class="tag">${t}</span>`).join('')
                        }</div>
                    </div>
                `;
            }

            if (result.context_chunks) {
                metadataHTML += `
                    <div class="metadata-item">
                        <div class="metadata-label">Context Chunks</div>
                        <div class="metadata-value">${result.context_chunks}</div>
                    </div>
                `;
            }

            metadataHTML += '</div>';

            resultItem.innerHTML = `
                <div class="result-header">
                    <div class="result-query">${escapeHtml(result.query)}</div>
                    <span class="quality-badge ${qualityClass}">${result.quality || 'unknown'}</span>
                </div>

                <div class="confidence-bar">
                    <div class="confidence-label">Confidence:</div>
                    <div class="bar">
                        <div class="bar-fill" style="width: ${confidencePercent}%"></div>
                    </div>
                    <div class="confidence-label">${confidencePercent}%</div>
                </div>

                <div class="result-answer">
                    ${escapeHtml(result.answer || 'No answer available')}
                </div>

                ${metadataHTML}

                ${result.top_result ? `
                    <div style="margin-top: 15px; padding: 12px; background: white; border-radius: 6px; border-left: 3px solid var(--primary);">
                        <div style="font-weight: 600; font-size: 12px; color: var(--gray-700); margin-bottom: 5px;">TOP RESULT</div>
                        <div style="font-size: 13px; color: var(--gray-900);">${escapeHtml(result.top_result)}</div>
                    </div>
                ` : ''}
            `;

            return resultItem;
        }

        // Show loading state
        function showLoading() {
            document.getElementById('loading').classList.add('active');
            document.getElementById('results-container').style.display = 'none';
            document.getElementById('empty-state').style.display = 'none';
        }

        // Hide loading state
        function hideLoading() {
            document.getElementById('loading').classList.remove('active');
        }

        // Show error message
        function showError(message) {
            const errorDiv = document.getElementById('error-message');
            errorDiv.textContent = message;
            errorDiv.classList.add('active');
        }

        // Hide error message
        function hideError() {
            const errorDiv = document.getElementById('error-message');
            errorDiv.classList.remove('active');
        }

        // Clear query
        function clearQuery() {
            document.getElementById('query-input').value = '';
            document.getElementById('priority-filter').value = '';
            document.getElementById('component-filter').value = '';
        }

        // Clear batch query
        function clearBatchQuery() {
            document.getElementById('batch-queries').value = '';
        }

        // Escape HTML
        function escapeHtml(text) {
            if (!text) return '';
            const map = {
                '&': '&amp;',
                '<': '&lt;',
                '>': '&gt;',
                '"': '&quot;',
                "'": '&#039;'
            };
            return text.replace(/[&<>"']/g, m => map[m]);
        }
    </script>
